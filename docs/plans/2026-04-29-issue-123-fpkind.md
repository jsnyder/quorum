```rust
/// Discriminates Verdict::Fp entries by reason. Calibrator applies different
/// decay/scope rules per kind. `Option<FpKind> = None` ↔ Hallucination
/// semantics — preserves behavior on pre-bump entries (no migration needed).
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
#[serde(rename_all = "snake_case")]
pub enum FpKind {
    /// LLM invented a defect. Structural; never expires.
    Hallucination,

    /// Real defect, but a compensating control prevents it.
    CompensatingControl { reference: String },

    /// FP under the current trust model only. 40d half-life (vs. 120d default).
    TrustModelAssumption,

    /// Pattern fires correctly elsewhere; THIS instance is the exception.
    /// Calibrator does NOT learn to blanket-suppress; discriminator hint
    /// goes into the few-shot prompt.
    PatternOvergeneralization { discriminator_hint: Option<String> },

    /// Real defect, tracked elsewhere. Excluded from the precedent pool.
    OutOfScope { tracked_in: Option<String> },
}
```

## Task 3: RED — Per-kind recency in `verdict_weight` (with clock injection refactor)

**Files:**
- Modify: `src/calibrator.rs` (function `verdict_weight` + all call sites + `mod tests`)

**Why clock injection:** the antipatterns review flagged that `chrono::Utc::now()` inline inside `verdict_weight` makes calibrator tests subtly time-coupled and prevents mutation testing from pinning the half-life constants. The refactor threads `now: DateTime<Utc>` through as a parameter; tests pin `now` exactly; production call sites pass `Utc::now()` once at the entry of `calibrate` / `calibrate_with_index` so all entries share the same `now`.

```rust
fn verdict_weight(entry: &FeedbackEntry, now: chrono::DateTime<chrono::Utc>) -> f64 {
    let provenance_weight = match &entry.provenance {
        crate::feedback::Provenance::PostFix => 1.5,
        crate::feedback::Provenance::Human => 1.0,
        crate::feedback::Provenance::External { .. } => 0.7,
        crate::feedback::Provenance::AutoCalibrate(_) => 0.5,
        crate::feedback::Provenance::Unknown => 0.3,
    };

    // Per-kind recency half-life (#123 Layer 1).
    // - TrustModelAssumption: 40d (3x faster decay) — these rot quickly.
    // - All other FpKind variants and non-FP verdicts: 120d default.
    // - Option<FpKind> = None for pre-bump entries: defaults to 120d
    //   (Hallucination semantics).
    let half_life_days = match (&entry.verdict, &entry.fp_kind) {
        (crate::feedback::Verdict::Fp, Some(crate::feedback::FpKind::TrustModelAssumption)) => 40.0,
        _ => 120.0,
    };
    let age_days = (now - entry.timestamp).num_days().unsigned_abs() as f64;
    let recency_weight = (-age_days / half_life_days).exp();

    provenance_weight * recency_weight
}
```

This guarantees all entries in a single calibration call share the same reference time.

```rust
struct FeedbackArgs {
    // existing fields...

    /// Discriminate FP verdict by kind (#123). Optional. One of:
    /// hallucination | trust-model | compensating-control |
    /// pattern-overgeneralization | out-of-scope.
    #[clap(long, value_parser = parse_fp_kind_variant)]
    fp_kind: Option<FpKindVariant>,

    /// Required with --fp-kind compensating-control. File:line, PR, or doc URL.
    #[clap(long, requires = "fp_kind")]
    fp_reference: Option<String>,

    /// Optional with --fp-kind pattern-overgeneralization. Hint surfaced in few-shot.
    #[clap(long, requires = "fp_kind")]
    fp_discriminator: Option<String>,

    /// Optional with --fp-kind out-of-scope. PR/issue tracking the real fix.
    #[clap(long, requires = "fp_kind")]
    fp_tracked_in: Option<String>,
}

#[derive(Clone, Debug, PartialEq, Eq, ValueEnum)]
enum FpKindVariant {
    Hallucination,
    TrustModel,
    CompensatingControl,
    PatternOvergeneralization,
    OutOfScope,
}

impl FeedbackArgs {
    fn into_fp_kind(&self) -> anyhow::Result<Option<FpKind>> {
        // Drop fp_kind when verdict != fp (composability with shell pipelines).
        // Caller emits tracing::warn at the call site so the user sees it.
        if self.verdict != Verdict::Fp {
            return Ok(None);
        }
        match self.fp_kind {
            None => Ok(None),
            Some(FpKindVariant::Hallucination) => Ok(Some(FpKind::Hallucination)),
            Some(FpKindVariant::TrustModel) => Ok(Some(FpKind::TrustModelAssumption)),
            Some(FpKindVariant::CompensatingControl) => {
                let reference = self.fp_reference.clone()
                    .ok_or_else(|| anyhow::anyhow!("--fp-kind compensating-control requires --fp-reference"))?;
                Ok(Some(FpKind::CompensatingControl { reference }))
            }
            Some(FpKindVariant::PatternOvergeneralization) => {
                Ok(Some(FpKind::PatternOvergeneralization {
                    discriminator_hint: self.fp_discriminator.clone(),
                }))
            }
            Some(FpKindVariant::OutOfScope) => {
                let tracked_in = self.fp_tracked_in.clone();
                if tracked_in.is_none() {
                    tracing::warn!(
                        "--fp-kind out-of-scope recorded without --fp-tracked-in; \
                         the deferral has no tracking link, future findings will keep firing"
                    );
                }
                Ok(Some(FpKind::OutOfScope { tracked_in }))
            }
        }
    }
}
```
